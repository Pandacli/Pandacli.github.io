根据开始时间 与结束时间 获取 一周 范围临时表

SELECT DATE_ADD(start_date, INTERVAL (t.a + u.a *10 + 100 * v.a) DAY) AS date_range
FROM (SELECT DATE_FORMAT(“2023-11-14 00:00:00”,’%Y-%m-%d’) AS start_date, DATE_FORMAT(“2023-12-01 00:00:00”,’%Y-%m-%d’) AS end_date) AS input
JOIN (SELECT 0 AS a UNION ALL SELECT 1 UNION ALL SELECT 2 UNION ALL SELECT 3 UNION ALL SELECT 4 UNION ALL SELECT 5 UNION ALL SELECT 6) AS t
JOIN (SELECT 0 AS a UNION ALL SELECT 1 UNION ALL SELECT 2 UNION ALL SELECT 3 UNION ALL SELECT 4 UNION ALL SELECT 5 UNION ALL SELECT 6) AS u
JOIN (SELECT 0 AS a UNION ALL SELECT 1 UNION ALL SELECT 2 UNION ALL SELECT 3 UNION ALL SELECT 4 UNION ALL SELECT 5 UNION ALL SELECT 6) AS v
WHERE DATE_ADD(start_date, INTERVAL (t.a + u.a *10 + 100 * v.a) DAY) <= end_date
ORDER BY date_range;
